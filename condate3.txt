    /* =============================================
       推奨PFC計算
    ============================================= */
    function calcRecommended(gender, weight) {
      const dailyKcal = gender === "male" ? weight * 33 : weight * 27;
      const mealKcal = Math.round(dailyKcal * 0.35);
      return {
        dailyKcal: Math.round(dailyKcal),
        kcal: mealKcal,
        protein: Math.round((mealKcal * 0.20) / 4),
        fat: Math.round((mealKcal * 0.25) / 9),
        carbs: Math.round((mealKcal * 0.55) / 4),
      };
    }

    /* =============================================
       食材マッチスコア (0〜1)
    ============================================= */
    function calcIngredientScore(recipe, userIngredients) {
      if (userIngredients.length === 0) return 1;
      const lowerUser = userIngredients.map(i => i.toLowerCase());
      const matched = recipe.mainIngredients.filter(ing =>
        lowerUser.some(u => ing.includes(u) || u.includes(ing))
      );
      return matched.length / recipe.mainIngredients.length;
    }

    /* PFCマッチスコア (0〜100) */
    function calcPFCScore(recipe, rec) {
      const kcalScore = Math.max(0, 100 - Math.abs(recipe.kcal - rec.kcal) / rec.kcal * 100);
      const totalRecKcal = rec.protein * 4 + rec.fat * 9 + rec.carbs * 4;
      const totalActKcal = recipe.protein * 4 + recipe.fat * 9 + recipe.carbs * 4;
      const pRec = (rec.protein * 4) / totalRecKcal;
      const fRec = (rec.fat * 9) / totalRecKcal;
      const cRec = (rec.carbs * 4) / totalRecKcal;
      const pAct = (recipe.protein * 4) / totalActKcal;
      const fAct = (recipe.fat * 9) / totalActKcal;
      const cAct = (recipe.carbs * 4) / totalActKcal;
      const pfcScore = Math.max(0, 100 - (Math.abs(pAct - pRec) + Math.abs(fAct - fRec) + Math.abs(cAct - cRec)) * 150);
      return Math.round(kcalScore * 0.5 + pfcScore * 0.5);
    }

    function PFCBar({ protein, fat, carbs }) {
      const total = protein * 4 + fat * 9 + carbs * 4;

          <div style={{ display: "flex", borderRadius: "8px", overflow: "hidden", height: "12px", marginBottom: "6px" }}>
            <div style={{ width: `${pPct}%`, background: "#4CAF50" }} />

    }

    function IngredientInput({ ingredients, setIngredients }) {

        const trimmed = val.trim();

          setIngredients(prev => [...prev, trimmed]);

        setInputVal("");

        if (e.key === "Enter" || e.key === "," || e.key === "　") {
          e.preventDefault();
          addIngredient(inputVal);

          setIngredients(prev => prev.slice(0, -1));
        }
      };

      ];

      const unusedSuggestions = SUGGESTIONS.filter(s => !ingredients.includes(s));

      return (

          {/* Tag input area */}
          <div
            onClick={() => inputRef.current?.focus()}
            style={{
              minHeight: "52px", padding: "8px 10px",

                  onClick={e => { e.stopPropagation(); setIngredients(prev => prev.filter(i => i !== ing)); }}
                  style={{
                    background: "rgba(255,255,255,0.3)", border: "none",
                    color: "white", borderRadius: "50%",
                    width: "18px", height: "18px", cursor: "pointer",

                onClick={() => { if (inputValid) { setStep("gacha"); setCurrent(null); setShowResult(false); setSpinCount(0); } }}
                disabled={!inputValid}
                style={{
                  width: "100%", padding: "13px 0",
                  background: inputValid ? "linear-gradient(135deg, #667eea, #764ba2)" : "#e0e0e0",
                  color: "white", border: "none", borderRadius: "14px",
                  fontSize: "15px", fontWeight: "bold",
                  cursor: inputValid ? "pointer" : "not-allowed",
                  boxShadow: inputValid ? "0 6px 20px rgba(102,126,234,0.45)" : "none",
                  transition: "all 0.2s", fontFamily: "inherit", letterSpacing: "1px",
                }}
              >
                {inputValid ? "次へ →" : "性別と体重を入力してください"}
              </button>
            </div>

            {/* ===== STEP 2: 食材入力 ===== */}
            <div style={{
              background: "white", borderRadius: "22px", padding: "22px 20px",
              boxShadow: "0 20px 60px rgba(0,0,0,0.28)", marginBottom: "16px",
            }}>
              <h2 style={{
                fontSize: "15px", fontWeight: "bold", color: "#444",
                margin: "0 0 14px", display: "flex", alignItems: "center", gap: "8px",
              }}>
                <span style={{
                  background: step === "gacha" ? "#4CAF50" : "#667eea",
                  color: "white", borderRadius: "50%", width: "24px", height: "24px",
                  display: "inline-flex", alignItems: "center", justifyContent: "center",
                  fontSize: "12px",
                }}>{step === "gacha" ? "✓" : "2"}</span>
                冷蔵庫の食材を入力
              </h2>

              <IngredientInput ingredients={ingredients} setIngredients={setIngredients} />

              {/* Match info */}
              {ingredients.length > 0 && step !== "gacha" && (
                <div style={{
                  marginTop: "12px", padding: "10px 14px",
                  background: isFiltered ? "#E8F5E9" : "#FFF8E1",
                  borderRadius: "12px",
                  border: `1px solid ${isFiltered ? "#4CAF50" : "#FFB300"}`,
                }}>
                  <span style={{ fontSize: "13px", color: isFiltered ? "#2E7D32" : "#F57F17", fontWeight: "bold" }}>
                    {isFiltered
                      ? `✅ ${matchingRecipes.length}種類のレシピが見つかりました！`
                      : `⚠️ マッチするレシピが少ないため全${ALL_RECIPES.length}種から引きます`}
                  </span>
                </div>
              )}
            </div>

            {/* ===== STEP 3: ガチャ ===== */}
            {step === "gacha" && rec && (
              <div style={{
                background: "white", borderRadius: "22px", padding: "22px 20px",
                boxShadow: "0 20px 60px rgba(0,0,0,0.28)",
              }}>
                <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: "14px" }}>
                  <h2 style={{
                    fontSize: "15px", fontWeight: "bold", color: "#444",
                    margin: 0, display: "flex", alignItems: "center", gap: "8px",
                  }}>
                    <span style={{
                      background: "#667eea", color: "white", borderRadius: "50%",
                      width: "24px", height: "24px",
                      display: "inline-flex", alignItems: "center", justifyContent: "center", fontSize: "12px",
                    }}>3</span>
                    献立ガチャ
                  </h2>
                  <button
                    onClick={() => { setStep("input"); setCurrent(null); setShowResult(false); }}
                    style={{
                      fontSize: "12px", color: "#667eea", background: "none",
                      border: "1.5px solid #667eea", borderRadius: "8px",
                      padding: "4px 12px", cursor: "pointer", fontFamily: "inherit",
                    }}
                  >← 戻る</button>
                </div>

                {/* Pool info */}
                <div style={{
                  padding: "8px 14px", marginBottom: "14px",
                  background: isFiltered ? "#E8F5E9" : "#F5F5F5",
                  borderRadius: "10px",
                  border: `1px solid ${isFiltered ? "#4CAF50" : "#e0e0e0"}`,
                  fontSize: "13px",
                  color: isFiltered ? "#2E7D32" : "#888",
                }}>
                  {isFiltered
                    ? `🎯 あなたの食材でできる ${gachaPool.length} 種類からガチャを引きます`
                    : `📦 全 ${ALL_RECIPES.length} 種類からガチャを引きます`}
                </div>

                {/* Initial state */}
                {!current && !isSpinning && (
                  <div style={{ textAlign: "center", padding: "28px 0" }}>
                    <div style={{ fontSize: "60px" }}>🎯</div>
                    <p style={{ color: "#888", fontSize: "14px", marginTop: "12px", lineHeight: 1.7 }}>
                      ガチャを回してあなたに<br />ぴったりの献立を引こう！
                    </p>
                  </div>
                )}

                {/* Recipe card */}
                {(current || isSpinning) && (
                  <div style={{
                    background: current?.bg || "#f9f9f9",
                    border: `3px solid ${current?.color || "#ddd"}`,
                    borderRadius: "18px", padding: "18px",
                    marginBottom: "16px", transition: "all 0.1s ease",
                  }}>
                    {/* Title row */}
                    <div style={{ display: "flex", alignItems: "center", gap: "12px", marginBottom: "12px" }}>
                      <div style={{ fontSize: "42px", lineHeight: 1 }}>{current?.emoji}</div>
                      <div style={{ flex: 1 }}>
                        <h3 style={{ margin: 0, fontSize: "18px", color: current?.color, fontWeight: "bold" }}>
                          {current?.name}
                        </h3>
                        {showResult && (
                          <div style={{ display: "flex", gap: "10px", marginTop: "4px" }}>
                            <span style={{ fontSize: "12px", color: "#666" }}>⏱ {current.time}</span>
                            <span style={{ fontSize: "12px", color: "#666" }}>{current.difficulty}</span>
                          </div>
                        )}
                      </div>
                      {showResult && pfcMatchScore !== null && <ScoreRing score={pfcMatchScore} />}
                    </div>

                    {showResult && (
                      <>
                        {/* Ingredient match */}
                        {ingredients.length > 0 && (
                          <div style={{ background: "white", borderRadius: "12px", padding: "10px 12px", marginBottom: "12px" }}>
                            <p style={{ fontSize: "12px", fontWeight: "bold", color: "#555", margin: "0 0 7px" }}>
                              🧊 食材マッチ（{Math.round(ingScore * 100)}%）
                            </p>
                            <IngredientMatchBadge recipe={current} userIngredients={ingredients} />
                          </div>
                        )}

                        {/* Calorie */}
                        <div style={{
                          display: "flex", alignItems: "center", justifyContent: "space-between",
                          background: "white", borderRadius: "12px", padding: "10px 14px",
                          marginBottom: "12px", border: `1.5px solid ${current.color}`,
                        }}>
                          <div>
                            <span style={{ fontSize: "26px", fontWeight: "bold", color: current.color }}>{current.kcal}</span>
                            <span style={{ fontSize: "13px", color: "#888", marginLeft: "4px" }}>kcal</span>
                          </div>
                          <div style={{ textAlign: "right" }}>
                            <div style={{ fontSize: "11px", color: "#aaa" }}>あなたの推奨</div>
                            <div style={{ fontSize: "16px", fontWeight: "bold", color: "#667eea" }}>{rec.kcal} kcal</div>
                          </div>
                        </div>

                        {/* PFC bar */}
                        <div style={{ background: "white", borderRadius: "12px", padding: "12px 14px", marginBottom: "12px" }}>
                          <p style={{ fontSize: "12px", fontWeight: "bold", color: "#555", margin: "0 0 8px" }}>
                            📊 PFCバランス
                          </p>
                          <PFCBar protein={current.protein} fat={current.fat} carbs={current.carbs} />
                        </div>

                        {/* Compare */}
                        <div style={{ background: "white", borderRadius: "12px", padding: "12px 14px", marginBottom: "12px" }}>
                          <p style={{ fontSize: "12px", fontWeight: "bold", color: "#555", margin: "0 0 10px" }}>
                            🔍 推奨値との比較
                          </p>
                          <CompareBar label="タンパク質(P)" recVal={rec.protein} actVal={current.protein} pfcColor="#4CAF50" />
                          <CompareBar label="脂質(F)" recVal={rec.fat} actVal={current.fat} pfcColor="#FF9800" />
                          <CompareBar label="炭水化物(C)" recVal={rec.carbs} actVal={current.carbs} pfcColor="#2196F3" />
                        </div>

                        <hr style={{ border: "none", borderTop: "1px dashed #ddd", margin: "12px 0" }} />

                        {/* Display ingredients */}
                        <p style={{ fontWeight: "bold", color: "#555", fontSize: "12px", margin: "0 0 7px" }}>🛒 使う食材</p>
                        <div style={{ display: "flex", flexWrap: "wrap", gap: "6px", marginBottom: "12px" }}>
                          {current.displayIngredients.map(ing => (
                            <span key={ing} style={{
                              background: "white",
                              border: `1px solid ${current.color}`,
                              color: current.color, borderRadius: "10px",
                              padding: "3px 10px", fontSize: "12px", fontWeight: "bold",
                            }}>{ing}</span>
                          ))}
                        </div>

                        {/* Steps */}
                        <p style={{ fontWeight: "bold", color: "#555", fontSize: "12px", margin: "0 0 8px" }}>📝 作り方</p>
                        {current.steps.map((s, i) => (
                          <div key={i} style={{ display: "flex", gap: "8px", marginBottom: "8px", alignItems: "flex-start" }}>
                            <span style={{
                              background: current.color, color: "white", borderRadius: "50%",
                              width: "20px", height: "20px",
                              display: "flex", alignItems: "center", justifyContent: "center",
                              fontSize: "11px", fontWeight: "bold", flexShrink: 0, marginTop: "1px",
                            }}>{i + 1}</span>
                            <span style={{ fontSize: "13px", color: "#444", lineHeight: 1.55 }}>{s}</span>
                          </div>
                        ))}
                      </>
                    )}
                  </div>
                )}

                {/* Spin button */}
                <button
                  onClick={spin}
                  disabled={isSpinning}
                  style={{
                    width: "100%", padding: "15px 0",
                    background: isSpinning ? "#ccc" : "linear-gradient(135deg, #f093fb 0%, #f5576c 100%)",
                    color: "white", border: "none", borderRadius: "50px",
                    fontSize: "17px", fontWeight: "bold",
                    cursor: isSpinning ? "not-allowed" : "pointer",
                    boxShadow: isSpinning ? "none" : "0 8px 24px rgba(245,87,108,0.45)",
                    transition: "all 0.2s", letterSpacing: "1px", fontFamily: "inherit",
                  }}
                >
                  {isSpinning ? "ガチャ中... 🎲" : showResult ? "もう一回ガチャ！🎰" : "献立ガチャを回す！🎰"}
                </button>

                {spinCount > 0 && !isSpinning && (
                  <p style={{ color: "#bbb", fontSize: "11px", textAlign: "center", marginTop: "8px" }}>
                    {spinCount}回目 ／ 対象{gachaPool.length}種類
                  </p>
                )}
              </div>
            )}
          </div>

          <p style={{ color: "rgba(255,255,255,0.5)", fontSize: "11px", marginTop: "24px", textAlign: "center" }}>
            ※ カロリー・栄養素は目安です。実際の量や調理法により異なります。
          </p>
        </div>
      );
    }

    const root = ReactDOM.createRoot(document.getElementById("root"));
    root.render(<App />);
  </script>
